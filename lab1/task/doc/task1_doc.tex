\begin{equation}
	\begin{split}
		\dot{\omega}l \left(1-\frac{m_s}{m_w + m_s} \cos(\phi)^2\right) = & g \sin(\phi) -  \frac{\cos(\phi)F}{m_w + m_s} - \frac{m_s}{m_w + m_s} {l \omega^2} \sin(\phi) \cos(\phi) \\
